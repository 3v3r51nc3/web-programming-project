\begin{frame}{Project Goal}
  \begin{exampleblock}{Main objective}
    Build \textbf{EventHub}, a responsive Single Page Application (SPA) for
    managing events, participants, and registrations.
  \end{exampleblock}

  \begin{itemize}
    \item Main backend: \textbf{Django REST API}
    \item Comparison backend: \textbf{Node.js / Express}
    \item Frontend: \textbf{React} with responsive HTML and CSS
    \item Team roles: Mehdi AGHAEI on frontend, Maksym DOLHOV on backend
    \item Final deliverable: one integrated repository and a deployable system
  \end{itemize}
\end{frame}

\begin{frame}{System Architecture}
  \begin{block}{Main flow}
    Browser $\rightarrow$ React SPA $\rightarrow$ Django REST API $\rightarrow$ Database
  \end{block}

  \vspace{0.4em}

  \begin{alertblock}{Comparison flow}
    Browser $\rightarrow$ React SPA $\rightarrow$ Node.js / Express backend
  \end{alertblock}

  \vspace{0.4em}

  \begin{itemize}
    \item React handles rendering, state, and API calls
    \item Django is the main structured backend
    \item Node.js / Express is kept for backend comparison
  \end{itemize}
\end{frame}

\begin{frame}{Backend Design}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Django}
        \begin{itemize}
          \item Models: \texttt{Event}, \texttt{Participant}, \texttt{Registration}
          \item Serializers convert data to JSON
          \item ViewSets expose CRUD endpoints
          \item Validation protects business rules
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Why Django first?}
        \begin{itemize}
          \item Strong structure
          \item Fast CRUD setup
          \item Good fit for relational data
          \item Admin and validation support
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}
