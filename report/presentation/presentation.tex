\begin{frame}{Project Overview}
  \begin{exampleblock}{Project goal}
    \textbf{EventHub} is a responsive event-management platform built to manage
    \textbf{events}, \textbf{participants}, and \textbf{registrations} inside one
    coherent SPA with a public landing page and a protected workspace.
  \end{exampleblock}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{What the project must show}
        \begin{itemize}
          \item Public home page plus protected user access
          \item CRUD workflows on real entities
          \item Clean frontend/backend separation
          \item Responsive UX on desktop and mobile
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Core entities}
        \begin{itemize}
          \item \texttt{Event}
          \item \texttt{Participant}
          \item \texttt{Registration}
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Main User Flow}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \begin{block}{Demo journey}
        \begin{enumerate}
          \item Open the public landing page at \texttt{/}
          \item Click a public event preview and trigger the sign-in warning
          \item Sign in through the retro auth modal
          \item Use the dashboard for a live project summary
          \item Create, edit, filter, and inspect events and participants with small pop-up windows
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.40\textwidth}
      \begin{exampleblock}{Visible product features}
        \begin{itemize}
          \item Event filtering by text, date, and status
          \item Theme toggle and consistent header controls
          \item Modal-based sign-in, register, create, and edit flows
          \item Loading, refresh, warning, and success states
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Architecture and Data Model}
  \begin{block}{Main system flow}
    Browser $\rightarrow$ React SPA $\rightarrow$ Django REST API $\rightarrow$ Database
  \end{block}

  \begin{alertblock}{Comparison backend}
    Node.js / Express is kept as a smaller comparison backend, while Django remains
    the main production-style backend for the project.
  \end{alertblock}

  \vspace{0.25em}

  \begin{tabularx}{\textwidth}{@{}lX@{}}
    \toprule
    \textbf{Entity} & \textbf{Role in the system} \\
    \midrule
    Event & stores title, date, location, description, and capacity \\
    Participant & stores attendee identity information \\
    Registration & explicit link between one participant and one event; enables validation and status tracking \\
    \bottomrule
  \end{tabularx}
\end{frame}

\begin{frame}{Frontend Implementation}
  \begin{block}{Reusable frontend structure}
    \begin{itemize}
      \item \texttt{components/common}: shared site frame and branding
      \item \texttt{components/auth}: login, register, social buttons, and auth modal
      \item \texttt{components/event}: CRUD, filters, cards, and registration widgets
      \item \texttt{components/header}, \texttt{pages}, \texttt{services}, \texttt{constants}, \texttt{utils}, and \texttt{styles}
    \end{itemize}
  \end{block}

  \begin{exampleblock}{Frontend engineering choices}
    \begin{itemize}
      \item One shared site frame for public and authenticated pages
      \item API calls isolated in a service layer
      \item Local storage for theme, session, and visitor counter behavior
      \item Responsive CSS and reusable UI states across pages
    \end{itemize}
  \end{exampleblock}
\end{frame}
