\begin{frame}{architecture}
  \begin{block}{main runtime path}
    browser / deployed frontend $\rightarrow$ react spa $\rightarrow$ django rest api $\rightarrow$ sqlite database
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \begin{exampleblock}{frontend $\leftrightarrow$ backend interaction}
        \begin{itemize}
          \item react calls the api through a centralized service layer
          \item jwt auth loads \texttt{/auth/token/}, refreshes tokens, and reads \texttt{/auth/me/}
          \item workspace data comes from \texttt{/events/}, \texttt{/participants/}, and \texttt{/registrations/}
          \item serializers and permissions decide what is saved and who can access it
        \end{itemize}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{alertblock}{comparison backend}
        express runs as a smaller separate api for comparison only.
        django is the main backend for auth, business rules, and registrations.
      \end{alertblock}

      \begin{beamercolorbox}[wd=\linewidth,sep=0.7em]{tagbox}
        \textbf{comparison routes}\\[0.25em]
        \texttt{:3001/api/events}\\
        \texttt{:3001/api/participants}
      \end{beamercolorbox}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{backend and database}
  \footnotesize
  \begin{columns}[T]
    \begin{column}{0.47\textwidth}
      \begin{block}{data model}
        \begin{itemize}
          \item \texttt{event}: title, date, location, capacity
          \item \texttt{participant}: first name, last name, email
          \item \texttt{registration}: event + participant + status
        \end{itemize}
        \vspace{0.35em}
        \centering
        \texttt{participant 1 --- n registration n --- 1 event}
      \end{block}

      \begin{alertblock}{why a separate registration model}
        the many-to-many link is not just a hidden join table. it has its own business meaning,
        its own status, and its own validation rules.
      \end{alertblock}
    \end{column}
    \begin{column}{0.49\textwidth}
      \begin{exampleblock}{business rules and api}
        \begin{itemize}
          \item duplicate registration is blocked in the backend
          \item capacity and past-date validation are enforced server-side
          \item admin can create, update, and delete
          \item viewer is read-only
          \item endpoints: \texttt{/events/}, \texttt{/participants/}, \texttt{/registrations/}
          \item auth: \texttt{/auth/register/}, \texttt{/auth/token/}, \texttt{/auth/me/}
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{django vs node.js}
  \small
  \begin{tabularx}{\textwidth}{@{}lXX@{}}
    \toprule
    \textbf{criteria} & \textbf{django} & \textbf{node.js / express} \\
    \midrule
    architecture & full framework with models, auth, serializers, admin, migrations & lightweight runtime + framework with more manual structure \\
    complexity & more initial structure, less custom wiring for this project & simpler start, but more manual validation and setup \\
    scalability & strong for structured business apps with permissions and data rules & flexible for smaller services and custom api layouts \\
    role in eventhub & main backend for auth, roles, registrations, and validation & comparison backend for smaller CRUD endpoints \\
    \bottomrule
  \end{tabularx}

  \vspace{0.55em}

  \begin{alertblock}{conclusion}
    django is the better main backend for eventhub because authentication,
    many-to-many registration logic, and permission rules are central to the system.
  \end{alertblock}
\end{frame}
