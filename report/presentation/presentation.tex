\begin{frame}{Project Overview}
  \small
  \begin{exampleblock}{Project goal}
    In this project, we built \textbf{EventHub}, a responsive web app for
    \textbf{events}, \textbf{participants}, and \textbf{registrations}. It combines
    a public landing page with a protected React workspace connected to a Django REST API.
  \end{exampleblock}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{What the project must show}
        \begin{itemize}
          \item Public event discovery and protected user access
          \item CRUD workflows on real database entities
          \item Clear separation between frontend, backend, and API
          \item Responsive behavior on desktop and mobile
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Core entities}
        \begin{itemize}
          \item \texttt{Event}: title, date, location, capacity
          \item \texttt{Participant}: identity and email
          \item \texttt{Registration}: event-participant link
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Main User Flow}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \begin{block}{Typical user flow}
        \begin{enumerate}
          \item Open the landing page and preview upcoming events
          \item Register or log in with the modal
          \item Enter the dashboard and load live API data
          \item Create, edit, filter, and inspect events and participants
          \item Manage registrations from the event details page
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.40\textwidth}
      \begin{exampleblock}{Visible features}
        \begin{itemize}
          \item Event filtering by text, date, and status
          \item Theme toggle and consistent navigation controls
          \item Modal-based sign-in, register, create, and edit flows
          \item Loading, refresh, warning, success, and error states
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Architecture and Data Model}
  \begin{block}{Main system flow}
    Browser $\rightarrow$ Vite / React SPA $\rightarrow$ Django REST API on \texttt{:8000} $\rightarrow$ SQLite database
  \end{block}

  \begin{alertblock}{Comparison backend}
    Node.js / Express runs separately on \texttt{:3001}. It is here for comparison,
    while Django remains the main backend of the project.
  \end{alertblock}

  \vspace{0.25em}

  \begin{tabularx}{\textwidth}{@{}lX@{}}
    \toprule
    \textbf{Entity} & \textbf{Role in the system} \\
    \midrule
    Event & stores title, description, date, location, capacity, and creation time \\
    Participant & stores attendee identity information and unique email \\
    Registration & links one participant to one event; enforces uniqueness, status, and capacity validation \\
    \bottomrule
  \end{tabularx}
\end{frame}

\begin{frame}{Frontend Implementation}
  \begin{block}{Reusable frontend structure}
    \begin{itemize}
      \item \texttt{components/common}: shared site frame, banners, and loading states
      \item \texttt{components/auth}: login, register, social buttons, and auth modal
      \item \texttt{components/event}: CRUD forms, filters, cards, and registration widgets
      \item \texttt{pages}, \texttt{services}, \texttt{constants}, \texttt{utils}, and \texttt{styles} separate screen, API, and helper logic
    \end{itemize}
  \end{block}

  \begin{exampleblock}{Frontend engineering choices}
    \begin{itemize}
      \item \texttt{VITE\_API\_BASE\_URL} controls which backend the SPA calls
      \item API calls are isolated in \texttt{src/services}
      \item Local storage keeps theme, session, and profile metadata
      \item Responsive CSS and reusable UI states keep the pages consistent
    \end{itemize}
  \end{exampleblock}
\end{frame}

\begin{frame}{Django vs Node.js / Express}
  \small
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Django}
        \begin{itemize}
          \item Full framework with models, serializers, auth, admin, and migrations
          \item Strong project structure and clear CRUD organization
          \item Better fit for EventHub because registrations and validation are central
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Node.js / Express}
        \begin{itemize}
          \item Node.js runtime with Express as a lightweight HTTP framework
          \item More flexible file structure, but more manual validation and setup
          \item Useful for smaller APIs and I/O-heavy services
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vspace{0.25em}

  \begin{alertblock}{Why both appear in this project}
    Django is the main backend because EventHub needs models, authentication,
    and registration logic. Express stays as the comparison backend to show the lighter approach.
  \end{alertblock}
\end{frame}
